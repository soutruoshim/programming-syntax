C Programming Cheatsheet

1. Basic Structure

#include <stdio.h>

int main() {
    printf("Hello, World!\n");
    return 0;
}

2. Data Types

TypeSize (Typical)Example
int4 bytesint a = 10;
float4 bytesfloat b = 5.5;
double8 bytesdouble c = 3.1415;
char1 bytechar ch = 'A';
void0 bytesvoid func();

3. Operators

Arithmetic: +, -, *, /, %

Relational: ==, !=, >, <, >=, <=

Logical: &&, ||, !

Bitwise: &, |, ^, ~, <<, >>

Assignment: =, +=, -=, etc.

Increment/Decrement: ++, --

4. Control Statements

if (a > b) {
    printf("a is greater\n");
} else {
    printf("b is greater\n");
}
for (int i = 0; i < 5; i++) {
    printf("%d\n", i);
}

int j = 0;
while (j < 5) {
    printf("%d\n", j);
    j++;
}

int k = 0;
do {
    printf("%d\n", k);
    k++;
} while (k < 5);

5. Functions

int add(int a, int b) {
    return a + b;
}

int main() {
    printf("%d", add(3, 4));
}

6. Arrays

int arr[3] = {1, 2, 3};
printf("%d", arr[0]);
int matrix[2][2] = {{1, 2}, {3, 4}};
printf("%d", matrix[1][1]);

7. Strings

char str[] = "Hello";
printf("%s", str);

Common Functions:

  • strlen(str)
  • strcpy(dest, src)
  • strcmp(str1, str2)
  • strcat(str1, str2)

8. Pointers

int a = 10;
int *p = &a;
printf("%d", *p);

9. Structures

struct Person {
    char name[20];
    int age;
};

struct Person p1 = {"Alice", 25};
printf("%s", p1.name);

10. File I/O

FILE *fp = fopen("file.txt", "w");
fprintf(fp, "Hello File!");
fclose(fp);
char buffer[100];
FILE *fp = fopen("file.txt", "r");
fgets(buffer, 100, fp);
printf("%s", buffer);
fclose(fp);

11. Dynamic Memory Allocation

int *ptr = (int *)malloc(5 * sizeof(int));
ptr[0] = 10;
free(ptr);

12. Preprocessor Directives

#define PI 3.14
#define MAX(a, b) ((a) > (b) ? (a) : (b))

13. Recursion

int factorial(int n) {
    if (n == 0) return 1;
    return n * factorial(n - 1);
}
