C++ Programming Cheatsheet

1. Basic Structure

#include <iostream>
using namespace std;

int main() {
    cout << "Hello, World!" << endl;
    return 0;
}

2. Data Types

TypeDescriptionExample
intIntegerint x = 10;
floatFloating pointfloat pi = 3.14f;
doubleDouble precisiondouble e = 2.71828;
charCharacterchar ch = 'A';
boolBooleanbool flag = true;
stringString (from <string>)string name = "Alice";

3. Operators

Arithmetic: +, -, *, /, %

Relational: ==, !=, >, <, >=, <=

Logical: &&, ||, !

Bitwise: &, |, ^, ~, <<, >>

Assignment: =, +=, -=, etc.

Increment/Decrement: ++, --

4. Control Flow

if (x > 10) {
    cout << "Greater than 10";
} else {
    cout << "10 or less";
}
for (int i = 0; i < 5; i++) {
    cout << i << endl;
}

int j = 0;
while (j < 5) {
    cout << j++ << endl;
}

do {
    cout << j++ << endl;
} while (j < 10);

5. Functions

int add(int a, int b) {
    return a + b;
}

7. Standard Template Library (STL)

#include <vector>
#include <map>
#include <algorithm>

vector<int> nums = {1, 2, 3};
map<string, int> ageMap;
ageMap["Alice"] = 30;
sort(nums.begin(), nums.end());

8. File I/O

#include <fstream>

ofstream fout("output.txt");
fout << "Hello File!";
fout.close();

ifstream fin("output.txt");
string line;
getline(fin, line);
cout << line;
fin.close();

9. Pointers and References

int x = 5;
int* p = &x;
int& ref = x;
cout << *p << ", " << ref;

10. Memory Management

int* ptr = new int;
*ptr = 10;
delete ptr;
